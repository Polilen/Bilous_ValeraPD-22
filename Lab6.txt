using System;
using System.Collections.Generic;

namespace lab6
{
    class Program
    {
        static void Main(string[] args)
        {
            int n;
            Console.Write("Введите номер: ");
            while(!int.TryParse(Console.ReadLine(), out n))
            {
                Console.Write("Вы ввели текст либо символ! Введите правильный номер: ");
            }
            string num = n.ToString();
            
            List<int> numList = new List<int>();
            for (int i = 0; i < num.Length; i++)
            {
                numList.Add(int.Parse(num[i].ToString()));
            }

            numList.Sort();
            int counter = 0;
            for (int i = 0; i < numList.Count; i++)
            {
                for (int j = 0; j < numList.Count; j++)
                {
                    if (numList[i] == numList[j])
                        counter++;
                }
                Console.WriteLine("Элемент {0} у нас здесь {1} раз!", numList[i], counter);
                i += counter - 1;
                counter = 0;
            }
            
            Console.Write("Введите номер, который вам надо найти: ");
            int numToFind;
            while (!int.TryParse(Console.ReadLine(), out numToFind))
            {
                Console.Write("Вы ввели текст либо символ! Введите правильный номер: ");
            }
            for (int i = 0; i < numList.Count; i++)
            {
                if (numList[i] == numToFind)
                {
                    Console.WriteLine("Ваш номер здесь в позиции {0}", i+1);
                    return;
                }
            }
            Console.WriteLine("В этом списке нет {0}!", numToFind);

        }
    }
}
